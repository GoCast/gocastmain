00037 /*
00038  * AnyOption 1.3  
00039  *
00040  * kishan at hackorama dot com  www.hackorama.com JULY 2001 
00041  *
00042  * + Acts as a common facade class for reading 
00043  *   commandline options as well as options from
00044  *   an optionfile with delimited type value pairs 
00045  *
00046  * + Handles the POSIX style single character options ( -w )
00047  *   as well as the newer GNU long options ( --width )
00048  * 
00049  * + The option file assumes the traditional format of
00050  *   first character based comment lines and type value
00051  *   pairs with a delimiter , and flags which are not pairs
00052  * 
00053  *      # this is a coment
00054  *      # next line is an option value pair
00055  *      width : 100
00056  *      # next line is a flag 
00057  *      noimages   
00058  * 
00059  * + Supports printing out Help and Usage  
00060  * 
00061  * + Why not just use getopt() ? 
00062  *
00063  *   getopt() Its a POSIX standard not part of ANSI-C. 
00064  *   So it may not be available on platforms like Windows.
00065  *
00066  * + Why it is so long ?
00067  *
00068  *   The actual code which does command line parsing 
00069  *   and option file parsing are done in  few methods. 
00070  *   Most of the extra code are for providing a flexible
00071  *   common public interface to both a resourcefile and
00072  *   and command line supporting POSIX style and  
00073  *   GNU long option as well as mixing of both. 
00074  * 
00075  * + Please see "anyoption.h" for public method descriptions 
00076  *   
00077  */

00652 AnyOption::processCommandArgs()
00653 {
00654         if( ! ( valueStoreOK() && CommandSet() )  )
00655            return;
00656            
00657         if( max_legal_args == 0 )
00658                 max_legal_args = argc;
00659         new_argv = (int*) malloc( (max_legal_args+1) * sizeof(int) );
00660         for( int i = 1 ; i < argc ; i++ ){/* ignore first argv */
00661                 if(  argv[i][0] == long_opt_prefix[0] && 
00662                      argv[i][1] == long_opt_prefix[1] ) { /* long GNU option */
00663                         int match_at = parseGNU( argv[i]+2 ); /* skip -- */
00664                         if( match_at >= 0 && i < argc-1 ) /* found match */
00665                                 setValue( options[match_at] , argv[++i] );

00667                         if( isPOSIX() ){

00669                                 if( ch != '0' && i < argc-1 ) /* matching char */
00670                                         setValue( ch ,  argv[++i] );
00671                         } else { /* treat it as GNU option with a - */
00672                                 int match_at = parseGNU( argv[i]+1 ); /* skip - */
00673                                 if( match_at >= 0 && i < argc-1 ) /* found match */
00674                                         setValue( options[match_at] , argv[++i] );
00675                         }
00676                 }else { /* not option but an argument keep index */
00677                         if( new_argc < max_legal_args ){
00678                                 new_argv[ new_argc ] = i ;
00679                                 new_argc++;
00680                         }else{ /* ignore extra arguments */
00681                                 printVerbose( "Ignoring extra argument: " );

00685                         }
00686                         printVerbose( "Unknown command argument option : " );

00721 int
00722 AnyOption::parseGNU( char *arg )
00723 {
00724         int split_at = 0;
00725         /* if has a '=' sign get value */
00726         for( unsigned int i = 0 ; i < strlen(arg) ; i++ ){
00727                 if(arg[i] ==  equalsign ){
00728                         split_at = i ; /* store index */
00729                         i = (unsigned int)strlen(arg); /* get out of loop */
00730                 }
00731         }
00732         if( split_at > 0 ){ /* it is an option value pair */
00733                 char* tmp = (char*) malloc(  (split_at+1)*sizeof(char) );
00734                 for( int i = 0 ; i < split_at ; i++ )
00735                         tmp[i] = arg[i];
00736                 tmp[split_at] = '\0';
00737 
00738                 if ( matchOpt( tmp ) >= 0 ){
00739                         setValue( options[matchOpt(tmp)] , arg+split_at+1 );
00740                         free (tmp);
00741                 }else{
00742                         printVerbose( "Unknown command argument option : " );
00743                         printVerbose( arg );
00744                         printVerbose( );
00745                         printAutoUsage();
00746                         free (tmp);
00747                         return -1;
00748                 }
00749         }else{ /* regular options with no '=' sign  */
00750                 return  matchOpt(arg);

01049  *
01050  *  assuming delimiter is ':' the behaviour will be,

01053  *  width : 10  - valid pair valuepairs( width, 10 );

01055  *  ::::        - not valid

01068         char *pline = (char*) malloc( (length+1)*sizeof(char) );
01069         for( int i = 0 ; i < length ; i ++ )
01070                 pline[i]= *(theline++);
01071         pline[length] = nullterminate;
01072         char *cursor = pline ; /* preserve the ptr */
01073         if( *cursor == delimiter || *(cursor+length-1) == delimiter ){
01074                 justValue( pline );/* line with start/end delimiter */
01075         }else{
01076                 for( int i = 1 ; i < length-1 && !found ; i++){/* delimiter */

01084                 cursor++;
